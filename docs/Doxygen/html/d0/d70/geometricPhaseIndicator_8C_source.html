104 void geometricPhaseIndicator::calcPhaseIndicator
105 (
106  volScalarField& alpha,
107  const volScalarField& psi
108 )
109 {
110  alpha == dimensionedScalar("0", dimless, 0);
111  const fvMesh& mesh = alpha.mesh();
112 
113  // Set phase-indicator to 1 and 0 in the bulk.
114  forAll(alpha, cellID)
115  {
116  if (psi[cellID] < 0)
117  alpha[cellID] = 1;
118  else
119  alpha[cellID] = 0;
120  }
121 
122  // Approach: Lookup existing narrowBand
123  const volScalarField& narrowBand = narrowBand_;
124 
125  // // Approch: Calc own narrowBand
126  // volScalarField& narrowBand = narrowBandTmp_.ref();
127  // const auto& own = mesh.owner();
128  // const auto& nei = mesh.neighbour();
129  // narrowBand = dimensionedScalar("narrowBand", narrowBand.dimensions(), 0.);
130 
131  // // Select the cells in the narrow band using face-connectivity.
132  // forAll(own, faceI)
133  // {
134  // // FIXME: If a cell is clipped by the interface at one of its corners,
135  // // and the surrounding cell centers do not switch the iso-sign, it
136  // // may not be picked up by this. May cause numerical inconsistency. TM.
137  // if (psi[own[faceI]] * psi[nei[faceI]] <= 0)
138  // {
139  // narrowBand[own[faceI]] = 1;
140  // narrowBand[nei[faceI]] = 1;
141  // }
142  // }
143  // // Set narrow band values across coupled process boundaries.
144  // const auto& psiBdryField = psi.boundaryField(); // needed for Nc LLSQ contribs
145  // const auto& patches = mesh.boundary(); // needed for Nc LLSQ contribs
146  // const auto& faceOwner = mesh.faceOwner();
147  // forAll(psiBdryField, patchI)
148  // {
149  // const fvPatch& patch = patches[patchI];
150  // if (isA<coupledFvPatch>(patch)) // coupled patch
151  // {

153  // auto psiPatchNeiFieldTmp =
154  // psiPatchField.patchNeighbourField();
155  // const auto& psiPatchNeiField = psiPatchNeiFieldTmp();
156  // forAll(psiPatchNeiField, faceI)
157  // {
158  // label faceJ = faceI + patch.start(); // Global face label.
159  // if (psi[faceOwner[faceJ]] * psiPatchNeiField[faceI] <= 0)
160  // {
161  // narrowBand[faceOwner[faceJ]] = 1;
162  // }
163  // }
164  // }
165  // }
166 
167  // Linear Least-Squares Approximation of \Psi
168  // \psi(x,y,z)= nc_x x + nc_y y + nc_z z + dc
169  volVectorField& nc_ = ncTmp_.ref();
170  nc_ = dimensionedVector("nc", nc_.dimensions(), vector(0,0,0));
171  volScalarField& dc_ = dcTmp_.ref();
172  dc_ = dimensionedScalar("dc", dc_.dimensions(), 0.);
173 
174  const auto& psiBdryField = psi.boundaryField();
175  const auto& patches = mesh.boundary();
176  const auto& cellCells = mesh.cellCells();
177  const auto& cellCenters = mesh.C();
178  // Compute the phase indicator in the narrow band.
179  forAll(narrowBand, cellI)
180  {
181  if (narrowBand[cellI] == 1)
182  {
183  // Assemble the LLSQ linear system
184  // TODO: Extend for 2D simulations in OpenFOAM.
185  //SquareMatrix<scalar> LLSQ(4,0);
186  simpleMatrix LLSQ
187  (
188  4 /* size 4x4 */,
189  0. /* init coeff value*/,
190  0. /* init source value*/
191  );
192  scalarList planeCoeffs(4,0);
193  scalarList LLSQsource(4,0);
194 
195  // NOTE: Nc doesn't contain cellI in OpenFOAM: cellI contributions
196  // are handled additionallly.
197  const auto& Nc = cellCells[cellI];
198 
199  // equations 0,1,2
200  // \frac{\partial e^{lsq}_c} {\partial n_{c,cmpt}} = 0
201  for (char row = 0; row < 3; ++row)
202  {
203  // Contributions from cellI : not in Nc
204  // - nc coefficient contrib from cellI
205  for(char col = 0; col < 3; ++col)
206  LLSQ(row,col) +=
207  cellCenters[cellI][col]*cellCenters[cellI][row];
208  // - dc coefficient contrib from cellI
209  LLSQ(row, 3) += cellCenters[cellI][row];
210  // - source contrib from cellI
211  LLSQsource[row] += psi[cellI]*cellCenters[cellI][row];
212 
213  // Contributions from Nc
214  forAll(Nc, cellK)
215  {
216  // - nc coefficient contrib from cellK
217  for(char col = 0; col < 3; ++col)
218  LLSQ(row,col) += cellCenters[Nc[cellK]][col]*cellCenters[Nc[cellK]][row];
219  // - dc coefficient contrib from cellK
220  LLSQ(row, 3) += cellCenters[Nc[cellK]][row];
221  // - source contrib from cellK
222  LLSQsource[row] += psi[Nc[cellK]]*cellCenters[Nc[cellK]][row];
223  }
224  }
225  // equation 3
226  // \frac{\partial e^{lsq}_c}{\partial d_c} = 0
227  // - nc coefficient contrib from cellI
228  for(char col = 0; col < 3; ++col)
229  LLSQ(3,col) += cellCenters[cellI][col];
230  // - source contrib for cellI
231  LLSQsource[3] += psi[cellI];
232 
233  // Contributions from Nc
234  forAll(Nc, cellK)
235  {
236  // - nc coefficient contrib from cellK
237  for(char col = 0; col < 3; ++col)
238  LLSQ(3,col) += cellCenters[Nc[cellK]][col];
239  // source term
240  LLSQsource[3] += psi[Nc[cellK]];
241  }
242  // - dc coefficient is |Nc|: in OpenFOAM it is |Nc| + 1 (for cellI)
243  LLSQ(3,3) = Nc.size() + 1;
244 
245  // Contributions from face-adjacent cells from coupled boundaries
246  const auto& nBandCell = mesh.cells()[cellI];
247  const auto& cellCentersBdryField = cellCenters.boundaryField();
248  forAll(nBandCell, faceI)
249  {
250  const label faceJ = nBandCell[faceI];
251  // If the face is not internal.
252  if (! mesh.isInternalFace(faceJ))
253  {
254  // Find the patch faceJ belongs to.
255  // - patch label.
256  label patchL = -1;
257  // - face index local to the patch
258  label faceP = -1;
259 
260  forAll(patches, patchI)
261  {
262  const fvPatch& patch = patches[patchI];
263  const polyPatch& ppatch = patch.patch();
264  faceP = ppatch.whichFace(faceJ);
265  // Patch is a coupled patch and the face faceP belongs to it.
266  if (isA<coupledFvPatch>(patch) && (faceP >= 0) && (faceP < patch.size()))
267  patchL = patchI;
268  }
269  // If faceJ (faceP) belongs to the patchL coupled patch.
270  if ((patchL != -1) && (faceP >= 0))
271  {
272  // The face belongs to the coupled boundary.
273  // Fetch the cell center of the cell.
274  const auto& cellCentersPatchField = cellCentersBdryField[patchL];
275  auto cellCentersPatchNeiFieldTmp = cellCentersPatchField.patchNeighbourField();
276  const auto& cellCentersPatchNeiField = cellCentersPatchNeiFieldTmp();
277  const auto& cellCenter = cellCentersPatchNeiField[faceP];
278 
279  // Fetch the psi value of the cell.
280  const auto& psiPatchField = psiBdryField[patchL];
281  auto psiPatchNeiFieldTmp = psiPatchField.patchNeighbourField();
282  const auto& psiPatchNeiField = psiPatchNeiFieldTmp();
283  const auto& psiValue = psiPatchNeiField[faceP];
284 
285  // equations 0,1,2 contrib from coupled faceJ-adjacent cell
286  // \frac{\partial e^{lsq}_c} {\partial n_{c,cmpt}} = 0
287  for (char row = 0; row < 3; ++row)
288  {
289  // - nc coefficient contrib from coupled faceJ-adjacent cell
290  for(char col = 0; col < 3; ++col)
291  LLSQ(row,col) += cellCenter[col]*cellCenter[row];
292 
293  // - dc coefficient contrib from coupled faceJ-adjacent cell
294  LLSQ(row, 3) += cellCenter[row];
295 
296  // - source contrib from coupled faceJ-adjacent cell
297  LLSQsource[row] += psiValue*cellCenter[row];
298  }
299 
300  // equation 3 contrib from coupled faceJ-adjacent cell
301  // \frac{\partial e^{lsq}_c}{\partial d_c} = 0
302 
303  // - nc coefficient contrib
304  for(char col = 0; col < 3; ++col)
305  LLSQ(3,col) += cellCenter[col];
306 
307  // - dc coeff contrib for faceJ (faceP) coupled neighbor-cell
308  LLSQ(3,3) += 1;
309 
310  // - source term contrib
311  LLSQsource[3] += psiValue;
312  }
313  }
314  }
315 
316  //LUscalarMatrix LU (LLSQ);
317  // Crashes in parallel, uses PStream::parRun() even if the linear
318  // system solution is local to the coupled process. TM.
319  //LU.solve(planeCoeffs, LLSQsource);
320  //Foam::LUsolve(LLSQ, LLSQsource);
321  scalarField& source = LLSQ.source();
322  source = LLSQsource;
323  planeCoeffs = LLSQ.solve(); // TODO: Improve this. Gauss substitution. TM.
324 
325  // Debugging
326  nc_[cellI] = vector(planeCoeffs[0], planeCoeffs[1], planeCoeffs[2]);
327  dc_[cellI] = planeCoeffs[3];
328 
329  // Ensure unit-normals
330  hesseNormalPlane cutPlane(
331  vector (planeCoeffs[0], planeCoeffs[1], planeCoeffs[2]),
332  planeCoeffs[3]
333  );
334 
335  // TODO: Use the cutPlane to test for intersection and do not
336  // intersect cells that do not require intersection. TM.
337  auto cellIntersection = intersectCell(cellI, mesh, cutPlane);
338 
339  // TODO: Check intersection when the plane overlaps with the face. TM.
340  // - Bounding cell intersection volume is a quick fix.
341  alpha[cellI] = max(cellIntersection.volume(), 0.) / mesh.V()[cellI];
342  }
343  }
344 
345  nc_.correctBoundaryConditions();
346  dc_.correctBoundaryConditions();
347  alpha.correctBoundaryConditions();
348 }
