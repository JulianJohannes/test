66 scalar maxDeltaT(surfaceScalarField phi, const dictionary& dict)
67 {
68  scalar maxGrowFactor = 1.2;
69  const fvMesh& mesh = phi.mesh();
70  scalar deltaT = mesh.time().deltaT().value();
71  scalar maxCFL = dict.get<scalar>("CFL");
72  scalarField sumPhi
73  (
74  fvc::surfaceSum(mag(phi))().primitiveField()
75  );
76 
77  scalar deltaT_suggestion = maxCFL / (0.5*gMax(sumPhi/mesh.V().field()));
78 
79  deltaT = min(deltaT * maxGrowFactor, deltaT_suggestion);
80 
81  return deltaT;
82 }
