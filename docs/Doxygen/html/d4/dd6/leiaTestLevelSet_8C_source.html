42 void redistancedPlaneIsUnchanged(autoPtr<redistancer>& redist, const fvMesh& mesh)
43 {
44  // TEST: Redistancing a signed distance from a plane has no
45  // effect if the redistancing algorithm is second-order
46  // accurate. Boundary values are calculated exactly.
47 
48  // Set psi to a signed distance from a plane.
49  const Time& runTime = mesh.time();
50  const volVectorField& xc = mesh.C();
51 
52  dimensionedVector pTest = Foam::average(xc);
53 
54  // TODO: extend the test with a parameterized normal orientation
55 
56  // Zero the plane-normal component orthogonal to the 2D solution plane.
57  // In the case of a 3D solution, all directions are 1.
58  vector nVec (1,2.1,3);
59  const auto& solutionDimensions = mesh.solutionD();
60  for (int i = 0; i < 3; ++i)
61  {
62  if (solutionDimensions[i] < 0)
63  nVec[i] = 0;
64  }
65  // Normalize to obtain signed-distance
66  nVec /= Foam::mag(nVec);
67  dimensionedVector nTest ("nTest", dimless, nVec);
68 
69  // Ignore boundary conditions for this test: signed-distances
70  // at the domain boundary are prescribed!
71  volScalarField psi
72  (
73  IOobject
74  (
75  "psi",
76  runTime.timeName(),
77  mesh,
78  IOobject::NO_READ,
79  IOobject::NO_WRITE
80  ),
81  mesh,
82  dimensionedScalar("0", dimLength, 0),
83  "calculated"
84  );
85  volScalarField psi0("psi0", psi);
86 
87  // Initialize signed distance to a plane
88  psi == (nTest & (xc - pTest));
89 
90  // Save the non-redistanced field for error evaluation
91  psi0 == psi;
92 
93  // |\nabla \psi0|
94  volScalarField magGradPsi0("magGradPsi0", Foam::mag(fvc::grad(psi0)));
95  Info << "mean(|\\nabla \\psi0|) : " <<
96  Foam::average(magGradPsi0).value() << endl;
97 
98  // ||\nabla \psi0| - 1|
99  volScalarField eGradPsi0 = Foam::mag(
100  Foam::mag(magGradPsi0) -
101  dimensionedScalar("1", dimless, 1)
102  );
103  eGradPsi0.rename("eGradPsi0");
104 
105  // Redistance the signed distance to a plane (nothing should happen)
106  redist->redistance(psi);
107 
108  // |\nabla \psi|
109  volScalarField magGradPsi("magGradPsi", Foam::mag(fvc::grad(psi)));
110  Info << "mean(|\\nabla \\psi|) : " <<
111  Foam::average(magGradPsi).value() << endl;
112 
113  volScalarField ePsi = Foam::mag(psi - psi0);
114  ePsi.rename("ePsi");
115 
116  // Valid errors available only at cell centers (internal field)
117  scalar ePsiLinf = Foam::max(ePsi).value();
118  if (ePsiLinf > 1e-13)
119  {
120  FatalErrorInFunction
121  << " redistancing a planar signed distance fails: \n"
122  << "Foam::max(Foam::mag(psi - psi0)) > 16*SMALL :"
123  << ePsiLinf
124  << abort(FatalError);
125  }
126 
127  // Valid errors available only at cell centers (internal field)
128  volScalarField eGradPsi = Foam::mag(
129  magGradPsi -
130  dimensionedScalar("1", dimless, 1)
131  );
132  eGradPsi.rename("eGradPsi");
133  eGradPsi.write();
134 
135  scalar eGradPsiLinf = Foam::max(eGradPsi).value();
136  Info << "eGradPsiLinf: " << eGradPsiLinf << endl;
137  if (eGradPsiLinf > 1e-11)
138  {
139  FatalErrorInFunction
140  << " redistancing a planar signed distance fails: \n"
141  << "Foam::max(Foam::mag(Foam::mag(fvc::grad(psi)) - 1) > 16*SMALL) :"
142  << eGradPsiLinf
143  << abort(FatalError);
144  }
145 
146  // TODO: Only if test fails.
147  Info << "Writing fields" << endl;
148 
149  psi0.write();
150  magGradPsi0.write();
151 
152  psi.write();
153  magGradPsi.write();
154 
155  ePsi.write();
156  eGradPsi0.write();
157  eGradPsi.write();
158 
159  Info << "Done." << endl;
160 };
