54 double Mollifier1::mollify(double x) const
55 {
56  double w1 = w1_;
57  double w2 = w2_;
58  // w1 is the width of the plateau
59  // w2 is the width before the mollifier is close to 0 (here 10^-3).
60  double s=1.0; // help variable
61  s = log(1000.0)/pow((w2-w1),2);
62  if (x >= 0) {
63  if (x < w1) {
64  return 1.0;
65  }
66  else {
67  return exp(- s*pow(x-w1, 2));
68  }
69  }
70  else {
71  return mollify(-x);
72  }
73 }
