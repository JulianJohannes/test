143  virtual void setVelocity(volVectorField& U) const
144  {
145  const fvMesh& mesh = U.mesh();
146  const auto& cellCenters = mesh.C();
147 
148  forAll(U, cellI)
149  {
150  U[cellI] = velocity(cellCenters[cellI]);
151  }
152 
153  const auto& Cf = mesh.Cf();
154  auto& UboundaryField = U.boundaryFieldRef();
155  const auto& CfBoundaryField = Cf.boundaryField();
156  const auto& meshBoundary = mesh.boundary();
157 
158  forAll(meshBoundary, patchI)
159  {
160  const auto& CfPatchField = CfBoundaryField[patchI];
161  auto& UpatchField = UboundaryField[patchI];
162  forAll(UpatchField, faceI)
163  {
164  UpatchField[faceI] = velocity(CfPatchField[faceI]);
165  }
166  }
167  }

169  virtual void setVolumetricFlux(surfaceScalarField& phi) const
170  {
171  const fvMesh& mesh = phi.mesh();
172  const surfaceVectorField& Sf = mesh.Sf();
173  const surfaceVectorField& Cf = mesh.Cf();
174 
175  forAll(Cf, faceID)
176  {
177  phi[faceID] = (velocity(Cf[faceID]) & Sf[faceID]);
178  }
179 
180  const auto& CfBoundaryField = Cf.boundaryField();
181  const auto& SfBoundaryField = Sf.boundaryField();
182  auto& phiBoundaryField = phi.boundaryFieldRef();
183  const auto& meshBoundary = mesh.boundary();
184 
185  forAll(meshBoundary, patchI)
186  {
187  const auto& CfPatchField = CfBoundaryField[patchI];
188  const auto& SfPatchField = SfBoundaryField[patchI];
189  auto& phiPatchField = phiBoundaryField[patchI];
190  forAll(phiPatchField, faceI)
191  {
192  phiPatchField[faceI] =
193  (
194  velocity(CfPatchField[faceI]) &
195  SfPatchField[faceI]
196  );
197  }
198  }
199  }

206 autoPtr<velocityModel> velocityModel::New(const fvMesh& mesh)
207 {
208  const fvSolution& fvSolution (mesh);
209  const dictionary& velocityDict = fvSolution.subDict("velocityModel");
210  const word modelType = velocityDict.getOrDefault<word>("type", "none");
211  Info<< "Selecting advection velocity " << modelType << endl;
212 
213  auto* ctorPtr = MeshConstructorTable(modelType);
214 
215  if (!ctorPtr)
216  {
217  FatalIOErrorInLookup
218  (
219  fvSolution,
220  "viscosityModel",
221  modelType,
222  *MeshConstructorTablePtr_
223  ) << exit(FatalIOError);
224  }
225 
226  return autoPtr<velocityModel>(ctorPtr(mesh));
227 }

289  //unsplit-advection volume tracking algorithm with planarity-preserving
290  //interface reconstruction. Computers and Fluids, 35(10), 1011–1032.

294 
295  public:

298  TypeName("shear3D");

301  :
302  velocityModel(mesh),
303  R(velocityDict_.getOrDefault<scalar>("R", 0.5)),
304  Umax(velocityDict_.getOrDefault<scalar>("Umax", 1.)),
305  x0(velocityDict_.getOrDefault<scalar>("x0", 0.5)),
306  y0(velocityDict_.getOrDefault<scalar>("y0", 0.5))
307  {};

309  virtual ~shear3D() = default;
310 
311  virtual vector velocity(const vector& p) const
312  {
313  const scalar& x = p[0];
314  const scalar& y = p[1];
315 
316  const scalar r = Foam::sqrt(
317  Foam::sqr(x - x0) +
318  Foam::sqr(y - y0)
319  );
320 
321  return vector(
322  Foam::sin(2*M_PI*y)*Foam::sqr(Foam::sin(M_PI * x)),
323  -Foam::sin(2*M_PI*x)*Foam::sqr(Foam::sin(M_PI * y)),
324  Umax * Foam::sqr(1 - (r / R))
325  );
326  }
327 };

458 void correctFlux(surfaceScalarField& phi)
459 {
460  const fvMesh& mesh = phi.mesh();
461  const Time& runTime = phi.time();
462 
463  volScalarField Evol = fvc::div(phi);
464 
465  Info << "gMax(fvc::div(phi)) = " << gMax(Evol) << endl;
466  Info << "gAverage(fvc::div(phi)) = " << gAverage(Evol) << endl;
467 
468  // Projection method enforcement of div(phi) = 0.
469  volScalarField p
470  (
471  IOobject
472  (
473  "p",
474  runTime.timeName(),
475  mesh,
476  IOobject::NO_READ,
477  IOobject::AUTO_WRITE
478  ),
479  mesh,
480  dimensionedScalar("p", dimArea / dimTime, 0),
481  "zeroGradient"
482  );
483 
484  // https://en.wikipedia.org/wiki/Projection_method_(fluid_dynamics)
485  // v = v_sol + v_irrot
486  // v = v_sol + \grad p / div
487  // \laplace p = \div v
488  // v_sol = v - \grad p
489  fvScalarMatrix pEqn
490  (
491  fvm::laplacian(p) == fvc::div(phi)
492  );
493  pEqn.solve();
494  phi = phi - pEqn.flux();
495  Evol = fvc::div(phi);
496 
497  Info << "Projection, gMax(fvc::div(phi)) = " << gMax(Evol) << endl;
498  Info << "Projection, gAverage(fvc::div(phi)) = " << gAverage(Evol) << endl;
499 
500  if (runTime.writeTime())
501  Evol.write();
502 
503 }
