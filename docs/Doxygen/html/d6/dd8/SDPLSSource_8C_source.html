106 tmp<fvScalarMatrix> SDPLSSource::fvmSDPLSSource(const volScalarField& psi, const volVectorField& U)
107 {
108  update(psi, U);
109  tmp<fvScalarMatrix> tfvm;
110  // if(mollifier_->TypeName() == "none")
111  if(mollifier_->type() == "none")
112  {
113  tfvm = discretization_->discretize(nonLinearPart_, psi);
114  }
115  else
116  {
117  volScalarField mollified_nonLinearPart = nonLinearPart_ * mollifier_->field(psi);
118  tfvm = discretization_->discretize(mollified_nonLinearPart, psi);
119  }
120 
121  return tfvm;
122 }

131 // old name 'minusR', new name 'R', wich mean the same term.
132 tmp<volScalarField> SDPLSSource::R(const volScalarField& psi, const volVectorField& U) const
133 {
134  volVectorField const grad_psi = gradPsi(psi);
135  volTensorField const grad_U = fvc::grad(U).cref();
136  dimensioned<scalar> const eps = dimensioned(grad_psi.dimensions(), SMALL);
137  volVectorField const normal_interface = (grad_psi / (mag(grad_psi) + eps)).cref();
138 
139  return (grad_U & normal_interface) & normal_interface;
140 }
