62 template<typename PointSequence>
63 scalar normalArea
64 (
65  PointSequence const& points,
66  const vector& normal
67 )
68 {
69  scalar area = 0;
70  using sizeType = decltype(points.size());
71  const auto size = points.size();
72  for (sizeType i = 0; i < size - 1; ++i)
73  {
74  const auto iNext = (i + 1) % size;
75  const point& ei = points[i];
76  const point& eiNext = points[iNext];
77  const vector edge = ei - eiNext;
78  vector edgeNormal = edge ^ normal;
79  edgeNormal /= Foam::mag(edgeNormal);
80  const vector edgeCenter = 0.5 * (ei + eiNext);
81  area += (edgeCenter & edgeNormal) * Foam::mag(edge);
82  }
83  area *= 0.5;
84  return Foam::mag(area);
85 }

99 template<typename PointSequence>
100 vector centroidAreaNormal
101 (
102  PointSequence const& points,
103  const point& centroid
104 )
105 {
106  vector areaNormal {0,0,0};
107 
108  using sizeType = decltype(points.size());
109  const auto size = points.size();
110  for (sizeType i = 0; i < size; ++i)
111  {
112  const auto iNext = (i + 1) % size;
113  areaNormal += 0.5 * ((points[i] - centroid) ^ (points[iNext] - centroid));
114  }
115 
116  return areaNormal;
117 }

119 template<typename OrientedPointSequence>
120 vector orientedAreaNormal
121 (
122  OrientedPointSequence const& points
123 )
124 {
125  vector areaNormal {0, 0, 0};
126  using sizeType = decltype(points.size());
127  const auto& xp0 = points[0];
128  for (sizeType p1 = 1; p1 < points.size(); ++p1)
129  {
130  sizeType p2 = (p1 + 1) % points.size();
131  areaNormal += 0.5*((points[p1] - xp0) ^ (points[p2] - xp0));
132  }
133  return areaNormal;
134 }

159 template<typename Intersection=volumeArea, typename ImplicitPlane>
160 Intersection intersectCell
161 (
162  label cellL,
163  const fvMesh& mesh,
164  ImplicitPlane plane
165 )
166 {
167  Intersection cellIntersection;
168 
169  // Unstructured mesh addressing
170  const auto& meshCells = mesh.cells();
171  const auto& meshFaces = mesh.faces();
172  const auto& faceCenters = mesh.Cf();
173  const auto& meshPoints = mesh.points();
174  const auto& faceOwner = mesh.faceOwner();
175 
176 #ifdef VISUALIZE_intersectCell
177  DynamicField<point> vtkPoints;
178  DynamicList<face> vtkFaces;
179  using Foam::surfaceWriters::vtkWriter;
180 #endif
181 
182  const auto& meshCell = meshCells[cellL];
183  std::vector<point> capPolygon;
184  capPolygon.reserve(4);
185  forAll(meshCell, faceL)
186  {
187  const label faceG = meshCell[faceL];
188  // Obtain the face centroid.
189  point xf {0, 0, 0};
190  if (mesh.isInternalFace(faceG))
191  {
192  xf = faceCenters[faceG];
193  }
194  else // Boundary face
195  {
196  const auto& faceCentersB = faceCenters.boundaryField();
197  const auto& patches = mesh.boundary();
198  // TODO: Improve pseudo 2D intersection, OpenFOAM. TM.
199  forAll(faceCentersB, patchI)
200  {
201  const fvPatch& patch = patches[patchI];
202  if (patch.type() == "empty") // 2D calculations on the "empty" patch
203  {
204  const face& emptyFace = mesh.faces()[faceG];
205  xf = point{0,0,0};
206  forAll(emptyFace, pointI)
207  {
208  xf += mesh.points()[emptyFace[pointI]];
209  }
210  xf /= emptyFace.size();
211  }
212  else if ((faceG >= patch.start()) &&
213  (faceG < patch.start() + patch.size()))
214  {
215  xf = faceCentersB[patchI][faceG - patch.start()];
216  }
217  }
218  }
219 
220  const auto& cellFace = meshFaces[faceG];
221  // OF-specific owner-neighbor face area-normal vector orientation
222  scalar orient = 1;
223  if (faceOwner[faceG] != cellL)
224  orient = -1;
225  for(label pointL = 0; pointL < cellFace.size(); ++pointL)
226  {
227  label pointLnext = (pointL + 1) % cellFace.size();
228 
229  const point& xfp0 = meshPoints[cellFace[pointL]];
230  const point& xfp1 = meshPoints[cellFace[pointLnext]];
231 
232  std::array<scalar,3> psiValues {0, 0, 0};
233  psiValues[0] = plane.value(xf);
234  psiValues[1] = plane.value(xfp0);
235  psiValues[2] = plane.value(xfp1);
236 
237  bool triangleOutside =
238  (psiValues[0] > 2*SMALL) &&
239  (psiValues[1] > 2*SMALL) &&
240  (psiValues[2] > 2*SMALL);
241 
242  if (!triangleOutside)
243  {
244  bool triangleInside =
245  (psiValues[0] < -2*SMALL) &&
246  (psiValues[1] < -2*SMALL) &&
247  (psiValues[2] < -2*SMALL);
248 
249  vector triAreaNormal = orient*0.5*((xfp0 - xf) ^ (xfp1 - xf));
250 
251  if (triangleInside)
252  {
253  cellIntersection.volume_ += (xf & triAreaNormal);
254 
255  #ifdef VISUALIZE_intersectCell // Visualize the inside triangle.
256  vtkFaces.append(
257  face {
258  vtkPoints.size(),
259  vtkPoints.size() + 1,
260  vtkPoints.size() + 2
261  }
262  );
263  vtkPoints.append(xf);
264  vtkPoints.append(xfp0);
265  vtkPoints.append(xfp1);
266  #endif
267  } else // The triangle is intersected
268  {
269  // Local addressing simplifies intersection.
270  std::array<point,3> triangle {xf, xfp0, xfp1};
271  if (orient == -1)
272  {
273  // Ensure the orientation of the local triangle
274  // generates outward-oriented area normal vector.
275  std::swap(triangle[1], triangle[2]);
276  std::swap(psiValues[1], psiValues[2]);
277  }
278  std::vector<point> triangleCut;
279  std::set<size_t> cutSet;
280  // Iterate over triangle points and intersect the triangle
281  for(label tp0 = 0; tp0 < 3; ++tp0)
282  {
283  label tp1 = (tp0 + 1) % 3;
284 
285  // Triangle point tP0 is inside \Omega^- but not a coplanar point
286  if (psiValues[tp0] < -2*SMALL)
287  {
288  triangleCut.push_back(triangle[tp0]);
289  }
290  // Triangle point tp0 is a coplanar and is an (intersection) point
291  if (Foam::mag(psiValues[tp0]) < 2*SMALL)
292  {
293  triangleCut.push_back(triangle[tp0]);
294  cutSet.insert(triangleCut.size() - 1);
295  }
296  else // Intersect the edge
297  {
298  sigFpe::unset(); // Disable OpenFOAM's SIGFPE trapping
299 
300  scalar lambda = psiValues[tp0] /
301  (psiValues[tp0] - psiValues[tp1]);
302 
303  //if (std::isnan(lambda)) // Both tp0 and tp1 are on the plane.
304  // If both tp0 and tp1 are on the plane.
305  if (std::isnan(lambda) ||
306  ((mag(psiValues[tp0]) <= 2*SMALL) &&
307  (mag(psiValues[tp0]) <= 2*SMALL)))
308  {
309  triangleCut.push_back(triangle[tp0]);
310  cutSet.insert(triangleCut.size() - 1);
311  }
312  else if ((lambda > 2*SMALL) && (lambda < 1-2*SMALL))
313  {
314  point cutPoint = triangle[tp0] +
315  lambda * (triangle[tp1] - triangle[tp0]);
316  triangleCut.push_back(cutPoint);
317  cutSet.insert(triangleCut.size() - 1);
318  }
319  sigFpe::set(); // Enable OpenFOAM's SIGFPE trapping
320  }
321 
322  }
323 
324  // Add contribution from the triangle cut polygon.
325  if (triangleCut.size() > 2)
326  {
327  cellIntersection.volume_ += triangleCut[0] & orientedAreaNormal(triangleCut);
328 
329  // Add oriented cut points to the cap polygon.
330  for(std::size_t tc0 = 0; tc0 < triangleCut.size(); ++tc0)
331  {
332  const std::size_t tc1 = (tc0 + 1) % triangleCut.size();
333  const bool tc0Found = cutSet.find(tc0) != cutSet.end();
334  const bool tc1Found = cutSet.find(tc1) != cutSet.end();
335 
336  if (tc0Found && tc1Found)
337  {
338  // Reverse cut point order to for outward orientation
339  // of the cap polygon.
340  capPolygon.push_back(triangleCut[tc1]);
341  capPolygon.push_back(triangleCut[tc0]);
342  }
343  }
344 
345  #ifdef VISUALIZE_intersectCell // Visualize the triangle cut polygon.
346  DynamicList<label> triangleCutFace;
347  std::size_t oldVtkPointsSize = vtkPoints.size();
348  for (std::size_t i = 0; i < triangleCut.size(); ++i)
349  {
350  triangleCutFace.append(oldVtkPointsSize + i);
351  vtkPoints.append(triangleCut[i]);
352  };
353  vtkFaces.append(face(triangleCutFace));
354  #endif
355  }
356 
357  }
358  }
359  }
360  }
361 
362  // Compute the area of the cap polygon.
363  scalar capPolygonArea = 0;
364  if (capPolygon.size() > 2)
365  {
366  // Coordinate system local to the cut triangle polygon.
367  // E01 coordinate system:
368  // Z-axis (k0dash) : reversed polygon normal
369  // Y-axis (j0dash) : first polygon edge
370  // X-axis (i0dash) : j0dash x k0dash
371 
372  // Cap polygon is stored as a sequcence of points
373  // but actually stores disconnected edges
374  // (c0, c1), (c2, c3), ...
375  // oriented counter-clockwise w.r.t. k0dash
376  vector k0dash = plane.normal();
377  k0dash /= mag(k0dash);
378  vector edge01 = capPolygon[1] - capPolygon[0];
379  scalar edge01mag = mag(edge01);
380  vector j0dash = edge01 / edge01mag;
381  vector i0dash = j0dash ^ k0dash;
382 
383  const auto& capPolygonSize = capPolygon.size();
384  using sizeType = decltype(capPolygon.size());
385  for(sizeType k = 0; k < capPolygonSize / 2; ++k)
386  {
387  sizeType e0 = 2*k;
388  sizeType e1 = (e0 + 1) % capPolygonSize;
389 
390  // Edge unit vector in global coordinates.
391  vector neDash = (capPolygon[e1] - capPolygon[e0]);
392  const scalar edgeMag = mag(neDash);
393  neDash /= edgeMag;
394  // Edge-normal unit vector in global coordinates.
395  neDash = neDash ^ k0dash;
396  // Edge-normal unit vector in local coordinates.
397  neDash = vector(
398  neDash & i0dash,
399  neDash & j0dash,
400  neDash & k0dash
401  );
402 
403  // First edge point in local coordinates.
404  vector xe0dash = capPolygon[e0] - capPolygon[0];
405  xe0dash = vector(
406  xe0dash & i0dash,
407  xe0dash & j0dash,
408  xe0dash & k0dash
409  );
410 
411  capPolygonArea += edgeMag * (neDash & xe0dash);
412  }
413  capPolygonArea *= 0.5; // Cap polygon is oriented into the cut volume.
414  // Add the volume contribution from the cap polygon.
415  cellIntersection.volume_ += capPolygon[0] & (k0dash * capPolygonArea);
416 
417  #ifdef VISUALIZE_intersectCell // Visualize the cap polygon.
418  DynamicList<label> capFace;
419  std::size_t oldVtkPointsSize = vtkPoints.size();
420  for (std::size_t i = 0; i < capPolygon.size(); ++i)
421  {
422  capFace.append(oldVtkPointsSize + i);
423  vtkPoints.append(capPolygon[i]);
424  }
425  vtkFaces.append(face(capFace));
426  #endif
427  }
428 
429  #ifdef VISUALIZE_intersectCell // Visualize the intersected cell
430  std::stringstream ss;
431  ss << "cutCell-" << std::setw(5) << std::setfill('0') << cellL;
432  vtkWriter writer(vtkPoints, vtkFaces, ss.str());
433  writer.write();
434  writer.close();
435  #endif
436 
437  cellIntersection.volume_ /= 3.;
438 
439  return cellIntersection;
440 };
