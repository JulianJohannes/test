73 bool Foam::functionObjects::gradPsiErrorCSV::write()
74 {
75 
76  const fvMesh& mesh = this->mesh_;
77  const Time& runTime = mesh.time();
78 
79  const auto deltaX = pow(mesh.deltaCoeffs(),-1).cref();
80  const auto max_deltaX = gMax(deltaX);
81  const auto mean_deltaX = gAverage(deltaX);
82  const auto gradPsiError = field_;
83  const auto max_gradPsiError = gMax(gradPsiError);
84  const auto mean_gradPsiError = gAverage(gradPsiError);
85  scalar max_narrowGradPsiError = 0.0;
86  scalar mean_narrowGradPsiError = 0.0;
87 
88  if (mesh.objectRegistry::found("NarrowBand"))
89  {
90  const auto narrowBand = mesh.lookupObject<volScalarField>("NarrowBand");
91  List<scalar> narrowGradPsiError = subset(narrowBand, gradPsiError);
92  max_narrowGradPsiError = gMax(narrowGradPsiError);
93  mean_narrowGradPsiError = gAverage(narrowGradPsiError);
94  }
95 
96  if (Pstream::myProcNo() == 0)
97  {
98  csvFile_ << runTime.timeOutputValue() << ","
99  << max_deltaX << ","
100  << mean_deltaX << ","
101  << max_gradPsiError << ","
102  << mean_gradPsiError << ","
103  << max_narrowGradPsiError << ","
104  << mean_narrowGradPsiError << "\n";
105  }
106  return true;
107 }
