42  void SignChangeNarrowBand::calc()
43  {
44  const auto& own = mesh().owner();
45  const auto& nei = mesh().neighbour();
46  const volScalarField& psi = this->psi();
47  volScalarField& NarrowBand = field();
48  NarrowBand = dimensionedScalar(NarrowBand.dimensions(), 0.);
49 
50  // Select the cells in the narrow band using face-connectivity.
51  forAll(own, faceI)
52  {
53  // FIXME: If a cell is clipped by the interface at one of its corners,
54  // and the surrounding cell centers do not switch the iso-sign, it
55  // may not be picked up by this. May cause numerical inconsistency. TM.
56  if (psi[own[faceI]] * psi[nei[faceI]] <= 0)
57  {
58  NarrowBand[own[faceI]] = 1;
59  NarrowBand[nei[faceI]] = 1;
60  }
61  }
62  // Set narrow band values across coupled process boundaries.
63  const auto& psiBdryField = psi.boundaryField(); // needed for Nc LLSQ contribs
64  const auto& patches = mesh().boundary(); // needed for Nc LLSQ contribs
65  const auto& faceOwner = mesh().faceOwner();
66  forAll(psiBdryField, patchI)
67  {
68  const fvPatch& patch = patches[patchI];
69  if (isA<coupledFvPatch>(patch)) // coupled patch
70  {
71  const auto& psiPatchField = psiBdryField[patchI];
72  auto psiPatchNeiFieldTmp =
73  psiPatchField.patchNeighbourField();
74  const auto& psiPatchNeiField = psiPatchNeiFieldTmp();
75  forAll(psiPatchNeiField, faceI)
76  {
77  label faceJ = faceI + patch.start(); // Global face label.
78  if (psi[faceOwner[faceJ]] * psiPatchNeiField[faceI] <= 0)
79  {
80  NarrowBand[faceOwner[faceJ]] = 1;
81  }
82  }
83  }
84  }
85  }
