92 scalar Foam::functionObjects::TVerrorCSV::calcTV()
93 {
94  const auto& own = mesh_.owner();
95  const auto& nei = mesh_.neighbour();
96  scalar TV = 0.0;
97  forAll(nei, faceI)
98  {
99  TV += mag(field_[nei[faceI]] - field_[own[faceI]]);
100  }
101 
102  // TV across coupled process boundaries.
103  const auto& boundaryField = field_.boundaryField();
104  const auto& patches = mesh_.boundary();
105  const auto& faceOwner = mesh_.faceOwner();
106  forAll(patches, patchI)
107  {
108  const fvPatch& patch = patches[patchI];
109  if (isA<coupledFvPatch>(patch)) // coupled patch
110  {
111  const auto& patchField = boundaryField[patchI];
112  auto patchNeiFieldTmp = patchField.patchNeighbourField();
113  const auto& patchNeiField = patchNeiFieldTmp();
114  forAll(patchNeiField, faceI)
115  {
116  label faceJ = faceI + patch.start(); // Global face label.
117  TV += mag(patchNeiField[faceJ] - field_[faceOwner[faceJ]]);
118  }
119  }
120  }
121 
122 
123  return TV;
124 }
