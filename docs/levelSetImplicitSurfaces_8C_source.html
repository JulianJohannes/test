421 scalar implicitSinc::value(const vector& x) const
422 {
423  double r = Foam::sqrt
424  (
425  Foam::sqr(x[0] - origin_[0]) +
426  Foam::sqr(x[1] - origin_[1])
427  );
428 
429  if (r < std::numeric_limits<double>::min())
430  return amplitude_;
431  else
432  {
433  return x[2] - origin_[2] - amplitude_ * sin(omega_ * r) / (omega_ * r);
434  }
435 }
